#include <Stepper.h>
#include <Wire.h>
#include <Servo.h>
const int stepsPerRevolution = 2048;  // change this to fit the number of steps per revolution
// for your motor

// initialize the stepper library on pins 2 through 5:
Stepper myStepper(stepsPerRevolution, 2, 4, 3, 5);

//Declare
Servo MyServo;

void setup() {
  // set the speed at 13 rpm:
  myStepper.setSpeed(13);
  MyServo.attach (6);
  // initialize the serial port:
  Serial.begin(9600);
}

void loop() {
  // step one revolution  in one direction:

  int x;
  
  for (x = 0; x !=2; x++)
  {
    
    myStepper.step(-stepsPerRevolution);
    Serial.println(x+1);
    delay(400);
  }
  x = 0;
  Serial.println("Delay in 4");
  MyServo.write(45);
  delay(1000);
  Serial.println("Delay in 3");
  MyServo.write(90);
  delay(1000);
  Serial.println("Delay in 2");
  MyServo.write(180);
  delay(1000);
  Serial.println("Delay in 1");
  MyServo.write(0);
  delay(1000);
}
